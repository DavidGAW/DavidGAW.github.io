微信小程序登录/验证登录/获取用户信息

在app.js中，微信小程序登录之后，为确保用户间隔性打开小程序都在登录状态下，所以在这里我把wx.login封装到一个函数里，在onLaunch中使用检验登录状态的wx.checkSession，如果session_key失效，再执行登录函数就可。在第三条写的是包含上述总结的app.js。
  1. 登录函数，发送code，换取openid，存在它，因后续的页面中的请求接口，需要传给接口；
  fun: function () {
    const urls = 'URL，也可以不在这赋给变量';
    // 登录
    wx.login({
    success: res => {
        // 发送 res.code 到后台换取 openId, sessionKey, unionId
        if (res.code) {
        //发起网络请求
        wx.request({
            url: urls + 'login',
            data: {
            code: res.code
            },
            success: function (res) {
            wx.setStorageSync('openid', res.data.token);
            }
        })
        } else {

        }
    }

    })

},
  1. 在onLaunch函数中 ，通过openid判断用户当前有没有登录，没有就调用登录方法，
onLaunch: function () {
    var that = this;
    if (wx.getStorageSync('openid')) {
        wx.checkSession({
            success: function () {
            //session_key 未过期，并且在本生命周期一直有效
            },
            fail: function () {
            // session_key 已经失效，需要重新执行登录流程
            this.fun() //重新登录
            }
        })
    } else {
    this.fun();
    }

},
  1. 在onLaunch函数中 ，wx.getSetting 回调中验证是否授权，如果授权过，将执行 wx.getUserInfo，获取用户信息，授权目前已更新到 需要button 触发，不支持弹出弹窗，按钮绑定 open-type=”getUserInfo” ，让用户自己选择，而不是一打开小程序 就弹出是否允许授权的窗口
wx.getSetting({

  success: res => {
    var that = this;
    if (res.authSetting['scope.userInfo']) {
      // 已经授权，可以直接调用 getUserInfo 获取头像昵称，不会弹框
      wx.getUserInfo({
        success: res => {
          // 可以将 res 发送给后台解码出 unionId
          this.globalData.userInfo = res.userInfo
          wx.setStorageSync("nickName", res.userInfo.nickName)
          wx.setStorageSync("avatarUrl", res.userInfo.avatarUrl)

          if (res.userInfo) {
            wx.request({
              url: '你的接口 用来发送到后台 验证授权',
              method: 'PUT',
              data: {
                nickName: wx.getStorageSync("nickName"),
                avatarUrl: wx.getStorageSync("avatarUrl"),
              },
              header: {
                'content-type': 'application/json',
                Token: wx.getStorageSync('openid'),
                Accept: 'application/vnd.epet.v1+json',
              },
              success: function (res) {

              },
              fail: function (res) { }
            });
          }
          // 由于 getUserInfo 是网络请求，可能会在 Page.onLoad 之后才返回
          // 所以此处加入 callback 以防止这种情况
          if (this.userInfoReadyCallback) {
            this.userInfoReadyCallback(res)
          }
        }
      })
    }
  }
})

onLaunch: function () {
    var that = this;
    if (wx.getStorageSync('openid')) {
    wx.checkSession({
        success: function () {
        //session_key 未过期，并且在本生命周期一直有效
        },
        fail: function () {

    this.fun();
