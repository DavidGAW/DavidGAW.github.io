Swiper 轮播图 也可做成跑马灯 就像信息提示/公告 在这里都是左右滚动

1.swiper 轮播图 wxml

<view class='lunBo'>
    <swiper class='swiper' indicator-dots='true' autoplay='true' interval='3000' duration=' 200' circular='true'>
        <block wx:for="{{lunBoimgUrls}}" wx:key="">
            <swiper-item>
                <image src="{{item}}" class="lunBo-image" />
            </swiper-item>
        </block>
    </swiper>
</view>

2.swiper 跑马灯 wxml

<!--跑马灯  -->
<view class='share-tips'>
    <view class='my-center'>
        <text class='tips-f'>公告：</text> 
        <swiper class='tips-k'  autoplay  interval='5000' duration='3000' circular='true' >
            <swiper-item wx:key="" wx:for='{{tipsText}}'>
                <view>{{item}}</view>
            </swiper-item>
        </swiper>
    </view>
</view>
